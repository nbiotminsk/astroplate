# Обязательно для правильной работы кодировки
AddDefaultCharset UTF-8

# Включаем механизм перенаправлений
<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /

    # 1. Принудительный HTTPS (SEO: Google любит безопасные сайты)
    RewriteCond %{HTTPS} off
    RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

    # 2. Перенаправление с WWW на без-WWW (SEO: избавляемся от дублей контента)
    # Например: www.meter.by -> meter.by
    RewriteCond %{HTTP_HOST} ^www\.(.*)$ [NC]
    RewriteRule ^(.*)$ https://%1/$1 [R=301,L]

    # 3. 301 редиректы со старых URL
    RewriteRule ^contacts/?$ /contact/ [R=301,L]
    RewriteRule ^price/?$ /services/ [R=301,L]

    # 4. Обеспечиваем работу PHP в папке api
    # Если запрос идет к PHP файлу, не трогаем его
    RewriteCond %{REQUEST_FILENAME} -f
    RewriteCond %{REQUEST_URI} \.php$
    RewriteRule ^ - [L]

    # 5. Убираем .html из конца URL (SEO: чистые и красивые адреса)
    RewriteCond %{THE_REQUEST} ^[A-Z]{3,9}\ /.*\.html\ HTTP/
    RewriteRule ^(.*)\.html$ /$1 [R=301,L]

    # 6. Обработка путей без расширения (Astro static)
    # Если файл существует, отдаем его
    RewriteCond %{REQUEST_FILENAME} -f
    RewriteRule ^ - [L]

    # Если это папка (например /services/), и в ней есть index.html, отдаем его
    RewriteCond %{REQUEST_FILENAME} -d
    RewriteCond %{REQUEST_FILENAME}/index.html -f
    RewriteRule ^(.*)$ $1/index.html [L]

    # Если файла нет, но есть файл с расширением .html
    RewriteCond %{REQUEST_FILENAME}.html -f
    RewriteRule ^(.*)$ $1.html [L]
</IfModule>

# Защита от кликджекинга и XSS (SEO: безопасность влияет на доверие поисковиков)
<IfModule mod_headers.c>
    Header set X-Content-Type-Options "nosniff"
    Header set X-Frame-Options "SAMEORIGIN"
    Header set X-XSS-Protection "1; mode=block"
    Header set Referrer-Policy "strict-origin-when-cross-origin"
</IfModule>

# Сжатие файлов (Gzip) для ускорения загрузки
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/plain
    AddOutputFilterByType DEFLATE text/html
    AddOutputFilterByType DEFLATE text/xml
    AddOutputFilterByType DEFLATE text/css
    AddOutputFilterByType DEFLATE application/xml
    AddOutputFilterByType DEFLATE application/xhtml+xml
    AddOutputFilterByType DEFLATE application/rss+xml
    AddOutputFilterByType DEFLATE application/javascript
    AddOutputFilterByType DEFLATE application/x-javascript
    AddOutputFilterByType DEFLATE image/svg+xml
</IfModule>

# Отключаем листинг файлов в папках (безопасность)
Options -Indexes

# Настройка кэширования для статики (1 год для медиа, неделя для стилей)
<IfModule mod_expires.c>
    ExpiresActive On
    ExpiresDefault "access plus 1 month"
    ExpiresByType image/x-icon "access plus 1 year"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/svg+xml "access plus 1 year"
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
</IfModule>
